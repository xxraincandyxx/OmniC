71// Internal function to link a new node between two existing nodes.
72static inline void _oc_list_link_node(oc_list_node_t* prev,
73 oc_list_node_t* next,
74 oc_list_node_t* new_node) {
75 new_node->next = next;
76 new_node->prev = prev;
77 if (prev) {
78 prev->next = new_node;
79 }
80 if (next) {
81 next->prev = new_node;
82 }
83}

85// Internal function to remove a node from the list.
