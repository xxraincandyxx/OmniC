73// Internal function to handle the growth logic.
74// Takes a void** so it can modify the user's original pointer variable.
75// Returns 0 on success, -1 on allocation failure.
76static inline int _oc_da_grow(void** da, size_t new_cap, size_t elem_size) {
77 // ... (function body remains unchanged)
78 size_t new_block_size = sizeof(oc_da_header_t) + (new_cap * elem_size);
79 oc_da_header_t* new_header = NULL;
80
81 if (*da == NULL) {
82 new_header = (oc_da_header_t*)malloc(new_block_size);
83 if (!new_header) {
84 return -1;
85 }
86 new_header->length = 0;
87 } else {
88 oc_da_header_t* old_header = _oc_da_header(*da);
89 new_header = (oc_da_header_t*)realloc(old_header, new_block_size);
90 if (!new_header) {
91 return -1;
92 }
93 }
94
95 new_header->capacity = new_cap;
96 // Point the user's dynarray pointer to the data section.
97 *da = (void*)((char*)new_header + sizeof(oc_da_header_t));
98 return 0;
99}

137#define oc_da_push(da, val) \
138 do { \
139 size_t cap = oc_da_cap(da); \
140 size_t len = oc_da_len(da); \
141 if (cap <= len) { \
142 size_t new_cap = (cap == 0) ? OC_DYNARRAY_INITIAL_CAPACITY : cap << 1; \
