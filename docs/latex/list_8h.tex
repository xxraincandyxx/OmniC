\doxysubsection{Detailed Description}
A generic, type-\/safe, macro-\/based intrusive doubly-\/linked list. 

This implementation uses an {\bfseries{"{}intrusive"{}}} design. This means you must embed an {\ttfamily \doxylink{structoc__list__node__t}{oc\+\_\+list\+\_\+node\+\_\+t}} member directly into the structure you wish to link. This approach is highly memory-\/efficient as it avoids separate allocations for list nodes.

{\bfseries{USAGE\+:}}

// 1. Define your data structure with an \doxylink{structoc__list__node__t}{oc\+\_\+list\+\_\+node\+\_\+t} member. typedef struct \{ int id; \doxylink{structoc__list__node__t}{oc\+\_\+list\+\_\+node\+\_\+t} link; // The intrusive node \} my\+\_\+item\+\_\+t;

// 2. Declare and initialize a list head. \doxylink{structoc__list__t}{oc\+\_\+list\+\_\+t} my\+\_\+list; \doxylink{list_8h_a963175ac7dd0f5b3288f7ae69b5452e6}{oc\+\_\+list\+\_\+init(\&my\+\_\+list)};

// 3. Add items to the list. oc\+\_\+list\+\_\+emplace\+\_\+back(\&my\+\_\+list, my\+\_\+item\+\_\+t, link, .id = 10); oc\+\_\+list\+\_\+emplace\+\_\+front(\&my\+\_\+list, my\+\_\+item\+\_\+t, link, .id = 5);

// 4. Iterate and use the list. my\+\_\+item\+\_\+t\texorpdfstring{$\ast$}{*} item = NULL; \doxylink{list_8h_a3ebea3df505bbd4723e97691c2ab1d53}{oc\+\_\+list\+\_\+for\+\_\+each(item, \&my\+\_\+list, my\+\_\+item\+\_\+t, link)} \{ printf("{}\+Item ID\+: \%d\textbackslash{}n"{}, item-\/\texorpdfstring{$>$}{>}id); \}
